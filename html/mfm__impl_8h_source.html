75 static mfm_io_symbol_t mfm_io_read_symbol(mfm_io_t *io) {
76  if (mfm_io_eof(io)) {
77  return mfm_io_pulse_10;
78  }
79  uint8_t pulse_len = io->pulses[io->pos++];
80  if (pulse_len > io->T3_max)
81  return mfm_io_pulse_1000;
82  if (pulse_len > io->T2_max)
83  return mfm_io_pulse_100;
84  return mfm_io_pulse_10;

138 static bool skip_triple_sync_mark(mfm_io_t *io) {
139  uint32_t state = 0;
140  while (!mfm_io_eof(io) && state != mfm_io_triple_mark_magic) {
141  state = ((state << 2) | mfm_io_read_symbol(io)) & mfm_io_triple_mark_mask;
142  }
143  DEBUG_PRINTF("mark @ %zd ? %d\n", io->pos, state == mfm_io_triple_mark_magic);
144  return state == mfm_io_triple_mark_magic;
145 }

156 // Copy 'n' bytes of data into 'buf'
157 static void receive(mfm_io_t *io, unsigned char *buf, size_t n) {
158  // `tmp` holds up to 9 bits of data, in bits 6..15.
159  unsigned tmp = 0, weight = 0x8000;
160 
161 #define PUT_BIT(x) \
162  do { \
163  if (x) \
164  tmp |= weight; \
165  weight >>= 1; \
166  } while (0)
167 
168  // In MFM, flux marks can be 2, 3, or 4 "T" apart. These three signals
169  // stand for the bit sequences 10, 100, and 1000. However, half of the
170  // bits are data bits, and half are 'clock' bits. We have to keep track of
171  // whether [in the next symbol] we want the "mfm_io_even" bit(s) or the
172  // "mfm_io_odd" bit(s):
173  //
174  // 10 - leaves mfm_io_even/mfm_io_odd (parity) unchanged
175  // 100 - inverts mfm_io_even/mfm_io_odd (parity)
176  // 1000 - leaves mfm_io_even/mfm_io_odd (parity) unchanged
177  // ^ ^ data bits if state is mfm_io_even
178  // ^ ^ data bits if state is mfm_io_odd
179 
180  // We do this by knowing that when we arrive, we are waiting to parse the
181  // final '1' data bit of the MFM sync mark. This means we apply a special rule
182  // to the first word, starting as though in the 'mfm_io_even' state but not
183  // recording the '1' bit.
184  mfm_io_symbol_t s = mfm_io_read_symbol(io);
185  mfm_state_t state = mfm_io_even;
186  switch (s) {
187  case mfm_io_pulse_100: // first data bit is a 0, and we start in the ODD state
188  state = mfm_io_odd;
189  /* fallthrough */
190  case mfm_io_pulse_1000: // first data bit is a 0, and we start in EVEN state
191  PUT_BIT(0);
192  break;
193  default: // this flux doesn't represent a data bit, and we start in the EVEN
194  // state
195  break;
196  }
197 
198  while (n) {
199  s = mfm_io_read_symbol(io);
200  PUT_BIT(state); // 'mfm_io_even' is 1, so record a '1' or '0' as appropriate
201  if (s == mfm_io_pulse_1000) {
202  PUT_BIT(0); // the other bit recorded for a 1000 is always a '0'
203  }
204  if (s == mfm_io_pulse_100) {
205  if (state) {
206  PUT_BIT(0);
207  } // If 'mfm_io_even', record an additional '0'
208  state = (mfm_state_t)!state; // the next symbol has opposite parity
209  }
210 
211  *buf = tmp >> 8; // store every time to make timing more mfm_io_even
212  if (weight <= 0x80) {
213  tmp <<= 8;
214  weight <<= 8;
215  buf++;
216  n--;
217  }
218  }
219 }

221 // Perform all the steps of receiving the next MFM_IO_IDAM, MFM_IO_DAM (or
222 // DMFM_IO_DAM, but we don't use them)
223 static bool skip_triple_sync_mark_receive_crc(mfm_io_t *io, void *buf,
224  size_t n) {
225  if (!skip_triple_sync_mark(io)) {
226  return false;
227  }
228  receive(io, (uint8_t *)buf, n);
229  unsigned crc = crc16_preloaded((uint8_t *)buf, n);
230  return crc == 0;
231 }

233 // Read a whole track, setting validity[] for each sector actually read, up to
234 // n_sectors indexing of validity & data is 0-based, mfm_io_even though
235 // MFM_IO_IDAMs store sectors as 1-based
236 static size_t decode_track_mfm(mfm_io_t *io) {
237  io->pos = 0;
238 
239  // count previous valid sectors, so we can early-terminate if we're just
240  // picking up some errored sectors on a 2nd pass
241  io->n_valid = 0;
242  for (size_t i = 0; i < io->n_sectors; i++)
243  if (io->sector_validity[i])
244  io->n_valid += 1;
245 
246  // Metadata structure is:
247  // * buf[0]: MFM_IO_IDAM
248  // * buf[1]: cylinder
249  // * buf[2]: head
250  // * buf[3]: sector
251  // * buf[4]: "n" (sector size shift)
252  // * buf[5:6]: crc
253  // Only the sector number is validated. In theory, the other values should be
254  // validated (but we don't have cylinder & head numbers at this layer) and we
255  // are only interested in working with DOS/Windows MFM floppies which always
256  // use 512 byte sectors
257  unsigned char buf[mfm_io_block_size + mfm_io_overhead];
258  while (!mfm_io_eof(io) && io->n_valid < io->n_sectors) {
259  if (!skip_triple_sync_mark_receive_crc(io, buf, mfm_io_idam_crc_size)) {
260  continue;
261  }
262  if (buf[0] != MFM_IO_IDAM) {
263  continue;
264  }
265 
266  // TODO: verify track & side numbers in IDAM
267  int r = (uint8_t)buf[3] - 1; // sectors are 1-based
268  if (r >= io->n_sectors) {
269  continue;
270  }
271 
272  if (io->sector_validity[r]) {
273  continue;
274  }
275 
276  // Sector structure is:
277  // * buf[0]: type (MFM_IO_DAM)
278  // * buf[1:513]: payload
279  // * buf[513:515]: crc16
280  if (!skip_triple_sync_mark_receive_crc(io, buf, sizeof(buf))) {
281  continue;
282  }
283  if (buf[0] != MFM_IO_DAM) {
284  continue;
285  }
286 
287  memcpy((char *)io->sectors + mfm_io_block_size * r, buf + 1,
288  mfm_io_block_size);
289  io->sector_validity[r] = 1;
290  io->n_valid++;
291  }
292  return io->n_valid;
293 }

300 void mfm_io_flux_byte(mfm_io_t *io, uint8_t byte) {
301  for (int i = 8; i-- > 0;) {
302  if (byte & (1 << i)) {
303  io->time += io->pulse_len + 1;
304  mfm_io_flux_put(io, (1 + io->pulse_len) * io->T1_nom);
305  io->pulse_len = 0;
306  } else {
307  io->pulse_len += 1;
308  }
309  }
310 }

312 void mfm_io_encode_raw(mfm_io_t *io, uint8_t byte) {
313  uint16_t y = (io->y << 8) | byte;
314  if ((byte & 0xaa) == 0) {
315  // if there are no clocks, synthesize them
316  y |= ~((y >> 1) | (y << 1)) & 0xaaaa;
317  y &= 0xff;
318  }
319  mfm_io_flux_byte(io, y);
320  io->y = y;
321 }

420 // Convert a whole track into flux, up to n_sectors. indexing of data is
421 // 0-based, mfm_io_even though MFM_IO_IDAMs store sectors as 1-based
422 static void encode_track_mfm(mfm_io_t *io) {
423  io->pos = 0;
424  io->pulse_len = 0;
425  io->y = 0;
426  io->time = 0;
427  // sector_validity might end up reused for interleave?
428  // memset(io->sector_validity, 0, io->n_sectors);
429 
430  unsigned char buf[mfm_io_idam_size];
431 
432  POS("pre-iam");
433  mfm_io_encode_iam(io);
434 
435  mfm_io_encode_gap_and_sync(io, mfm_io_gap_1);
436  for (size_t i = 0; i < io->n_sectors; i++) {
437  POS("pre-sector %zd", i);
438  buf[0] = MFM_IO_IDAM;
439  POS("idam payload");
440  buf[1] = io->cylinder;
441  buf[2] = io->head;
442  buf[3] = i + 1; // sectors are 1-based
443  buf[4] = mfm_io_n;
444 
445  mfm_io_crc_preload(io);
446  mfm_io_encode_buf_crc(io, buf, sizeof(buf));
447  mfm_io_encode_crc(io);
448 
449  mfm_io_encode_gap_and_sync(io, mfm_io_gap_2);
450  mfm_io_crc_preload(io);
451  mfm_io_encode_byte_crc(io, MFM_IO_DAM);
452  POS("dam payload");
453  mfm_io_encode_buf_crc(io, &io->sectors[mfm_io_block_size * i], 512);
454  mfm_io_encode_crc(io);
455 
456  mfm_io_encode_gap_and_sync(io, mfm_io_gap_3);
457  }
458  assert(!mfm_io_eof(io));
459 
460  while (!mfm_io_eof(io)) {
461  mfm_io_encode_byte(io, MFM_IO_GAP_BYTE);
462  }
463 }
